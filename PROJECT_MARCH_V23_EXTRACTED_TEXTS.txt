function getRandomParanoiaThought() {
  if (!PARANOIA_THOUGHTS.length) return '';
  let idx = Math.floor(Math.random() * PARANOIA_THOUGHTS.length);
  if (PARANOIA_THOUGHTS.length > 1) {
    while (idx === paranoiaLastThoughtIndex) idx = Math.floor(Math.random() * PARANOIA_THOUGHTS.length);
  }
  paranoiaLastThoughtIndex = idx;
  return PARANOIA_THOUGHTS[idx];
}

  if (paranoiaBubbleTimer) clearTimeout(paranoiaBubbleTimer);

  bubble.classList.remove('show');
  void bubble.offsetWidth;
  bubble.classList.add('show');
  const duration = Math.max(2200, Math.min(3600, 1200 + text.length * 38));

}

    return '';

  paranoiaThoughtCounter += 1;

function applyParanoiaDelta(delta) {
  let total = delta;
  let burst = false;
  if (delta > 0) {
    paranoiaRaiseStreak += 1;
    if (paranoiaRaiseStreak >= 2) {

      burst = true;

  evham = Math.max(0, Math.min(10, evham + total));

function applyHiddenEffect(card, side) {
  if(!card || !card.hidden) return;

function advanceSpecialAfterNormalTurn() {

  const readyIndex = pendingSpecialQueue.findIndex(item => item.normalsLeft <= 0);

}

  }
}

  }, 700);
}
